# GStreamer QTI Allocator base library.
set(GST_QTI_ALLOCATOR_BASE gstqtiallocatorsbase)

add_library(${GST_QTI_ALLOCATOR_BASE} SHARED
  gstqtiallocator.c
)

set(GST_QTI_ALLOCATOR_BASE_HEADERS
  gstqtiallocator.h
)

set_target_properties(${GST_QTI_ALLOCATOR_BASE} PROPERTIES PUBLIC_HEADER
  "${GST_QTI_ALLOCATOR_BASE_HEADERS}"
)

target_compile_definitions(${GST_QTI_ALLOCATOR_BASE} PRIVATE
  $<$<BOOL:${HAVE_LINUX_DMA_HEAP_H}>:HAVE_LINUX_DMA_HEAP_H>
)

target_include_directories(${GST_QTI_ALLOCATOR_BASE} PUBLIC
  ${GST_INCLUDE_DIRS}
)

target_include_directories(${GST_QTI_ALLOCATOR_BASE} PRIVATE
  ${KERNEL_BUILDDIR}/usr/include
)

target_link_libraries(${GST_QTI_ALLOCATOR_BASE} PRIVATE
  ${GST_LIBRARIES}
  ${GST_ALLOC_LIBRARIES}
)

install(
  TARGETS ${GST_QTI_ALLOCATOR_BASE}
  LIBRARY DESTINATION ${GST_PLUGINS_QTI_OSS_INSTALL_LIBDIR}
  PERMISSIONS OWNER_EXECUTE OWNER_WRITE OWNER_READ
              GROUP_EXECUTE GROUP_READ
              WORLD_EXECUTE WORLD_READ
  PUBLIC_HEADER DESTINATION ${GST_PLUGINS_QTI_OSS_INSTALL_INCDIR}/gstreamer-1.0/gst/allocators/
)
